class record_mode(object):
  def __enter__(self):
    cfg.__CONDENSA_RECORD_MODE__ = True

  def __exit__(self, *args):
    cfg.__CONDENSA_RECORD_MODE__ = False
    return False

    """Constructs an `LC` class instance.

       :param steps: Number of L-C iterations.
       :type steps: float
       :param l_batch_size: Batch size for L-step.
       :type l_batch_size: int
       :param lr: Starting learning rate.

       :param lr_end: Ending learning rate.
       :type lr_end: float
       :param lr_schedule: Learning rate schedule.
       :type lr_schedule: List
       :param lr_multiplier: Learning rate multiplier.
       :type lr_multiplier: float
       :param momentum: Momentum for L-step.
       :type momentum: float
       :param weight_decay: Weight decay for L-step.
       :type weight_decay: float
       :param sgd_iterations_per_l: Number of SGD iterations per L-step.
       :type sgd_iterations_per_l: int
       :param sgd_iterations_first_l: Number of SGD iterations for first L-step.
       :type sgd_iterations_first_l: int
       :param mu_init: Initial value of `mu`.
       :type mu_init: float
       :param mu_multiplier: Mu multiplier.
       :type mu_multiplier: float
       :param mu_cap: Maximum permitted value for `mu`.
       :type mu_cap: float
       :param val_batch_size: Validation/test batch size.
       :type val_batch_size: int
    """
    self._engine_config = {k:v for k,v in locals().items() if k != 'self'}
    logger.info('[Condensa] LC ENGINE CONFIG [' + ', '.join('{!s}={!r}'
                .format(k,v) for k,v in self._engine_config.items())
                + ']')

    self.use_cuda = torch.cuda.is_available()
    self.steps = steps
    self.l_batch_size = l_batch_size
    self.lr = lr
    self.lr_end = lr_end
    self.lr_decay = lr_decay
    self.lr_schedule = lr_schedule
    if self.lr_schedule is not None and lr_multiplier is None:
      raise TypeError('Please specify multiplier when using fixed LR schedule')
    self.lr_multiplier = lr_multiplier
    self.momentum = momentum
    self.weight_decay = weight_decay
    self.sgd_iterations_per_l = sgd_iterations_per_l
    self.sgd_iterations_first_l = sgd_iterations_first_l
    self.mu_init = mu_init
    self.mu_multiplier = mu_multiplier
    self.mu_cap = mu_cap
    self.val_batch_size = val_batch_size



[docs]  def condensa_loss(self, mu, w, theta, lm):
    """Computes L-C reconstruction loss.
    
       :param mu: LC `mu` hyper-parameter value.
       :type mu: float
       :param w: Input model.
       :type w: torch.nn.Module
       :param theta: Compressed model.
       :type theta: torch.nn.Module
       :param lm: Lagrange multiplier.
       :type lm: torch.nn.Module
    """
    w.eval()
    loss = 0.
    with torch.no_grad():
      for w_m, theta_m, lm_m in zip(w.modules(), theta.modules(), lm.modules()):
        if hasattr(theta_m, 'condense'):
          for pname in theta_m.condense:
            w_p = getattr(w_m, pname).detach()
            theta_p = getattr(theta_m, pname).data
            lm_p = getattr(lm_m, pname).data
            compression_loss = w_p-theta_p
            qp_term = (compression_loss*compression_loss).sum()
            al_term = (compression_loss*lm_p).sum()
            loss += 0.5 * mu * qp_term - al_term
    return loss



[docs]  def zero_(self, model):
    """Zeroes out model parameters.
    
       :param model: PyTorch model.
       :type model: torch.nn.Module
    """
    with torch.no_grad():
      pflat = torch.nn.utils.parameters_to_vector(model.parameters()).fill_(0.)
      torch.nn.utils.vector_to_parameters(pflat, model.parameters())



[docs]  def compress(self,
               w,
               pi,
               delta,
               trainset,
               testset,
               valset,
               loss_fn):
    """Main L-C compression method.
    
       :param w: Input model.
       :type w: torch.nn.Module
       :param pi: Compression function.
       :param delta: Decompression function.
       :param trainset: Training dataset.
       :param testset: Test dataset.
       :param valset: Validation dataset.
       :param loss_fn: Loss criterion.
    """

    statistics = {}
    # Save engine configuration
    statistics.update(self._engine_config)

    timer_lc = EventTimer()

    validate = (valset is not None)
    test = (testset is not None)
    trainloader = data.DataLoader(trainset,
                                  batch_size=self.l_batch_size,
                                  shuffle=True,
                                  pin_memory=True,
                                  num_workers=8)
    if test:
      testloader = data.DataLoader(testset,
                                   batch_size=self.val_batch_size,
                                   shuffle=False,
                                   pin_memory=True,
                                   num_workers=8)
    if validate:
      valloader = data.DataLoader(valset,
                                  batch_size=self.val_batch_size,
                                  shuffle=False,
                                  pin_memory=True,
                                  num_workers=8)

    # Copy model to GPU memory
    if self.use_cuda: w = w.cuda()

    # Mark all compressible modules in w
    with record_mode(): pi(w)

    with torch.no_grad(): theta = deepcopy(w)
    self.zero_(theta)

    with torch.no_grad(): lm = deepcopy(w)
    self.zero_(lm)

    with torch.no_grad(): best_model = deepcopy(w)

    # Enable data-parallelism in  L step
    if self.use_cuda: w = torch.nn.DataParallel(w)

    mu = 0.
    learning_rate = self.lr

    optimizer = LCSGD(w, lr=learning_rate, momentum=self.momentum)
    optimizer.zero_velocities()

    w_train_loss, w_train_acc = loss_and_accuracy(
      w, loss_fn, trainloader)
    logger.info('[Condensa] w TRAIN\tloss={:.5f}, acc={:.5f}'
                .format(w_train_loss, w_train_acc))
    if validate:
      w_val_loss, w_val_acc = loss_and_accuracy(
        w, loss_fn, valloader)
      logger.info('[Condensa] w VAL\tloss={:.5f}, acc={:.5f}'
                  .format(w_val_loss, w_val_acc))
    if test:
      w_test_loss, w_test_acc = loss_and_accuracy(
        w, loss_fn, testloader)
      logger.info('[Condensa] w TEST\tloss={:.5f}, acc={:.5f}'
                  .format(w_test_loss, w_test_acc))

    best_acc = 0.
    train_accuracies = []
    train_losses = []
    if validate:
      val_accuracies = []
      val_losses = []
    test_accuracies = []
    test_losses = []
    outer_lr_scheduler = None
    if self.lr_decay is not None:
      outer_lr_scheduler = ExpDecayedLR(self.lr, self.lr_decay)
    elif self.lr_schedule is not None:
      outer_lr_scheduler = DecayedLR(self.lr,
                                     self.lr_schedule,
                                     self.lr_multiplier)
    for j in range(0, self.steps):
      #w_train_loss, w_train_full_loss, w_train_acc = self.stats(
      #  mu, w, theta, lm, loss_fn, trainloader)
      #w_test_loss, w_test_full_loss, w_test_acc = self.stats(
      #  mu, w, theta, lm, loss_fn, testloader)
      #logger.info('[Condensa] w TRAIN\tloss={:.5f}, condensa_loss={:.5f}, '
      #            'acc={:.5f}'
      #            .format(w_train_loss, w_train_full_loss, w_train_acc))
      #logger.info('[Condensa] w TEST\tloss={:.5f}, condensa_loss={:.5f}, '
      #            'acc={:.5f}'
      #            .format(w_test_loss, w_test_full_loss, w_test_acc))

      n_sgd_iter = (self.sgd_iterations_first_l if j == 1
                    else self.sgd_iterations_per_l)

      # Set up outer learning rate
      learning_rate = self.lr
      if outer_lr_scheduler is not None:
        learning_rate = outer_lr_scheduler.learning_rate

      logger.info('[Condensa] LC Iteration {}:\tmu={:.5f}, lr={:.5f}'
                  .format(j, mu, learning_rate))

      inner_lr_scheduler = None
      if self.lr_end is not None:
        #n = self.sgd_iterations_per_l if j>1 else self.sgd_iterations_first_l
        #inner_lr_scheduler = IntervalLR(self.lr, self.lr_end, n)
        inner_lr_scheduler = IntervalLR(learning_rate, self.lr_end, n_sgd_iter)

      # L step
      # Switch to training mode
      i = 0
      mb_iterator = iter(trainloader)
      elapsed_dataload = 0.
      elapsed_zero = 0.
      elapsed_step = 0.
      timer_l = EventTimer()
      while True:
        if j == 0:
          logger.info('[Condensa] Skipping first L-step')
          break

        timer_dataload = EventTimer()
        try:
          inputs, targets = next(mb_iterator)
        except StopIteration:
          mb_iterator = iter(trainloader)
          inputs, targets = next(mb_iterator)

        w.train()
        if self.use_cuda:
          inputs, targets = inputs.cuda(), targets.cuda()
        inputs, targets = torch.autograd.Variable(inputs), torch.autograd.Variable(targets)
        elapsed_dataload += timer_dataload.elapsed_seconds
        outputs = w(inputs)
        loss = loss_fn(outputs, targets)

        timer_zero = EventTimer()
        optimizer.zero_grad()
        elapsed_zero += timer_zero.elapsed_seconds

        loss.backward()
        timer_step = EventTimer()
        optimizer.step(learning_rate, mu, theta, lm)
        elapsed_step += timer_step.elapsed_seconds

        #learning_rate *= l_alpha
        if inner_lr_scheduler is not None:
          inner_lr_scheduler.step()
          learning_rate = inner_lr_scheduler.learning_rate

        if j == 1 and i > self.sgd_iterations_first_l: break
        if j > 1 and i > self.sgd_iterations_per_l: break
        i += 1

        if logger.isEnabledFor(logging.INFO):
          sys.stdout.write('\r[Condensa] >> ')
          sys.stdout.write('[%.0f%%] L STEP (/%d)' % (100. * float(i) / float(n_sgd_iter), n_sgd_iter))
          sys.stdout.flush()

      logger.info('')

      w.eval()
      # C step and theta update
      try: theta.load_state_dict(w.module.state_dict())
      except AttributeError: theta.load_state_dict(w.state_dict())
      if mu > 0:
        try: wmodules = w.module.modules()
        except AttributeError: wmodules = w.modules()
        with record_mode(): pi(theta)
        with torch.no_grad():
          for w_m, theta_m, lm_m in zip(wmodules,
                                        theta.modules(),
                                        lm.modules()):
            if hasattr(theta_m, 'condense'):
              for pname in theta_m.condense:
                getattr(theta_m, pname).data = (getattr(w_m, pname).detach()
                                                - getattr(lm_m, pname).data/mu)



      pi(theta)

      nested_train_loss, nested_train_acc = loss_and_accuracy(
        theta, loss_fn, trainloader)
      logger.info('[Condensa] Nested (theta) TRAIN\tloss={:.5f}, acc={:.5f}'
        .format(nested_train_loss, nested_train_acc))

      if validate:
        nested_val_loss, nested_val_acc = loss_and_accuracy(
          theta, loss_fn, valloader)
        logger.info('[Condensa] Nested (theta) VAL\tloss={:.5f}, acc={:.5f}'
          .format(nested_val_loss, nested_val_acc))

      if test:
        nested_test_loss, nested_test_acc = loss_and_accuracy(
          theta, loss_fn, testloader)
        logger.info('[Condensa] Nested (theta) TEST\tloss={:.5f}, acc={:.5f}'
          .format(nested_test_loss, nested_test_acc))

      train_losses.append(nested_train_loss)
      train_accuracies.append(nested_train_acc)
      if test:
        test_losses.append(nested_test_loss)
        test_accuracies.append(nested_test_acc)
      if validate:
        val_losses.append(nested_val_loss)
        val_accuracies.append(nested_val_acc)

      if validate:
        if nested_val_acc > best_acc:
          logger.info('[Condensa] SAVING MODEL based on VAL')
          best_acc = nested_val_acc
          # Deep-copy required here to preserve dtypes
          best_model = deepcopy(theta)
      elif test:
        if nested_test_acc > best_acc:
          logger.info('[Condensa] SAVING MODEL based on TEST')
          best_acc = nested_test_acc
          # Deep-copy required here to preserve dtypes
          best_model = deepcopy(theta)
      else:
        logger.info('[Condensa] SAVING MODEL based on most recent')
        best_model = deepcopy(theta)

      # theta <- delta(theta)
      delta(theta)

      # LM update
      if mu > 0:
        try: wmodules = w.module.modules()
        except AttributeError: wmodules = w.modules()
        for w_m, theta_m, lm_m in zip(wmodules,
                                      theta.modules(),
                                      lm.modules()):
          if hasattr(theta_m, 'condense'):
            for pname in theta_m.condense:
              getattr(lm_m, pname).data = (getattr(lm_m, pname).data
                - mu*(getattr(w_m, pname).detach()
                - getattr(theta_m, pname).data))


      optimizer.zero_velocities()
      # Update mu
      mu = self._update_mu(mu, self.mu_init, self.mu_multiplier, self.mu_cap)
      # Update LR schedule
      if outer_lr_scheduler is not None: outer_lr_scheduler.step()

      #print('Dataload: ', elapsed_dataload)
      #print('Zero: ', elapsed_zero)
      #print('Step: ', elapsed_step)

    statistics['elapsed_lc'] = timer_lc.elapsed_seconds
    statistics['train_losses'] = train_losses
    statistics['train_accuracies'] = train_accuracies
    statistics['test_losses'] = test_losses
    statistics['test_accuracies'] = test_accuracies
    if validate:
      statistics['val_losses'] = val_losses
      statistics['val_accuracies'] = val_accuracies
    return best_model, statistics


  def _update_mu(self, mu, mu_init, mu_multiplier, mu_cap):
    if mu > mu_cap:
      return mu
    if mu != 0:
      return mu * mu_multiplier
    else:
      return mu_init
